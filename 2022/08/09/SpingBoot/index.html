多数据源配置

数据源信息配置

spring:
  datasource:
      # 数据源一
    olddm:
      jdbc-url: jdbc:mysql://localhost:3306/olddm
      uesrname: oldblack
      password: YB520lly..
      driver-class-name: com.mysql.cj.jdbc.Driver
    # 数据源二
    dbuser:
      jdbc-url: jdbc:mysql://localhost:3306/db_user
      uesrname: oldblack
      password: YB520lly..
      driver-class-name: com.mysql.cj.jdbc.Driver

在application.yml文件中配置需要连接的数据源。如上配置了olddm和dbuser两个数据源。

产生同数据源对应的DataSourceBean

@Configuration
public class DataSourceConfig {

    @Bean
    @ConfigurationProperties(prefix = "spring.datasource.olddm")
    public DataSource olddmDataSource() {
        return DataSourceBuilder.create().build();
    }

    @Bean
    @ConfigurationProperties(prefix = "spring.datasource.dbuser")
    public DataSource dbUserDataSource() {
        return DataSourceBuilder.create().build();
    }
}

使用@ConfigurationProperties注入不同的数据源信息，@Bean注解生产对应的Bean对象。

简单多数据源配置（使用JdbcTemplate）

  • 依赖
//jdbc
implementation 'org.springframework.boot:spring-boot-starter-jdbc:2.7.3
//数据库连接jar
runtimeOnly 'mysql:mysql-connector-java'
  • JdbcTemplateBean配置
@Configuration
public class JdbcTemplateConfig {

    @Bean
    public JdbcTemplate olddmJdbcTemplate(
            @Qualifier("olddmDataSource") DataSource dataSource) {
        return new JdbcTemplate(dataSource);
    }


    @Bean
    public JdbcTemplate dbUserJdbcTemplate(
            @Qualifier("dbUserDataSource") DataSource dataSource) {
        return new JdbcTemplate(dataSource);
    }
}

@Qualifier(“olddmDataSource”)实现方法中参数需要的数据源Bean注入。也可以使用@Resource等注解在属性上注入。

Mybatis整合多数据源

  • 依赖
//jdbc
implementation 'org.springframework.boot:spring-boot-starter-jdbc:2.7.3
//数据库连接jar
runtimeOnly 'mysql:mysql-connector-java'
//mybatis plus
implementation 'com.baomidou:mybatis-plus-boot-starter:3.5.2'
  • 配置mapper的位置、实体类位置
mybatis-plus:
  mapper-locations:
      # 数据源一的mapper位置
    db-user: classpath:mapper/dbuser/*.xml
    # 数据源二的mapper位置
    olddm: classpath:mapper/olddm/*.xml
  type-aliases-package: com.mds.domain
  • 实现每个数据眼的Config，生成对应的SqlSessionFactory和SqlSessionTemplate
@Configuration
@MapperScan(
        basePackages = "com.mds.dao.olddm",
        sqlSessionFactoryRef = "olddmSqlSessionFactory",
        sqlSessionTemplateRef = "olddmSqlSessionTemplate"
)
public class OlddmDSConfig {

//    注入对应的数据源
    private DataSource olddmDataSource;

//    对应数据源的mapper映射文件位置
    @Value("${mybatis-plus.mapper-locations.olddm}")
    private String olddmMapperLocation;

    public OlddmDSConfig(@Qualifier("olddmDataSource")
                         DataSource olddmDataSource) {
        this.olddmDataSource = olddmDataSource;
    }

    @Bean
    public SqlSessionFactory olddmSqlSessionFactory() throws Exception {
        SqlSessionFactoryBean sqlSessionFactoryBean = new SqlSessionFactoryBean();
        sqlSessionFactoryBean.setDataSource(this.olddmDataSource);
//        加载映射文件
        Resource[] resource = new PathMatchingResourcePatternResolver().getResources(
                this.olddmMapperLocation
        );
        sqlSessionFactoryBean.setMapperLocations(
                resource
        );
        return sqlSessionFactoryBean.getObject();
    }

    @Bean
    public SqlSessionTemplate olddmSqlSessionTemplate() throws Exception {
        return new SqlSessionTemplate(olddmSqlSessionFactory());
    }

}

@Configuration
@MapperScan(
        basePackages = "com.mds.dao.dbuser",
        sqlSessionFactoryRef = "dbUserSqlSessionFactory",
        sqlSessionTemplateRef = "dbUserSqlSessionTemplate"
)
public class UserDSConfig {

    private DataSource dbUserDataSource;

    @Value("${mybatis-plus.mapper-locations.db-user}")
    private String dbUserMapperLocation;

    public UserDSConfig(@Qualifier("dbUserDataSource")
                        DataSource dbUserDataSource) {
        this.dbUserDataSource = dbUserDataSource;
    }

    @Bean
    public SqlSessionFactory dbUserSqlSessionFactory() throws Exception {
        SqlSessionFactoryBean sqlSessionFactoryBean = new SqlSessionFactoryBean();
        sqlSessionFactoryBean.setDataSource(this.dbUserDataSource);
        Resource[] resource = new PathMatchingResourcePatternResolver().getResources(
                this.dbUserMapperLocation
        );
        sqlSessionFactoryBean.setMapperLocations(
                resource
        );
        return sqlSessionFactoryBean.getObject();
    }

    @Bean
    public SqlSessionTemplate dbUserSqlSessionTemplate() throws Exception {
        return new SqlSessionTemplate(dbUserSqlSessionFactory());
    }

}
