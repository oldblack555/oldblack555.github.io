定时任务

使用方式

spring定时任务使用@Scheduled注解声明对应的方法，需要使用@EnableScheduling注解开启定时任务

@Async
@Scheduled(initialDelay = TimerDelay.HALF_DAY, fixedDelay = TimerDelay.ONE_DAY)
public void allUpdate() {
    List<Long> dmids = tbDmStatusMapper.queryUpdate();
    detailTimer(dmids);
}

//方式一
//@EnableScheduling注解结合Configuration注解使用进行打开定时任务
@Configuration
@EnableAsync
@EnableScheduling
public class SpiderTimer {}

//方式二
//在springboot启动类上声明@EnableScheduling注解
@SpringBootApplication
@EnableScheduling
public class OlddmApplication {}

属性

属性 说明
fixedRate 以固定速率执行，即每隔时间t执行一次
fixedDelay 上次执行完毕后延迟再执行，即本次执行完毕后等待时间t再次执行
initialDelay 第一次延时执行，常与fixedRate、fixedDelay搭配使用，即延时t后第一次执行任务，第一次结束后按照fixedRate或fixedDelay的规则执行，否则以后的效果与fixedDelay一样
cron 按cron规则执行
  • cron常用表达式
表达式 说明
0 0 2 1 * ? * 表示在每月 1 日的凌晨 2 点执行
0 15 10 ? * MON-FRI 表示周一到周五每天上午 10:15 执行
0 15 10 ? 6L 2019-2020 表示 2019-2020 年的每个月的最后一个星期五上午 10:15 执行
0 0 10,14,16 * * ? 每天上午 10 点，下午 2 点，4 点执行
0 0/30 9-17 * * ? 朝九晚五工作时间内每半小时执行
0 0 12 ? * WED 表示每个星期三中午 12 点执行
0 0 12 * * ? 每天中午 12点执行
0 15 10 ? * * 每天上午 10:15 执行
0 15 10 * * ? 每天上午 10:15 执行
0 15 10 * * ? * 每天上午 10:15 执行
0 15 10 * * ? 2019 2019 年的每天上午 10:15 执行

ApplicationContextAware接口实现全局Context获取

package com.dm.config;

import org.jetbrains.annotations.NotNull;
import org.springframework.beans.BeansException;
import org.springframework.context.ApplicationContext;
import org.springframework.context.ApplicationContextAware;
import org.springframework.context.annotation.Configuration;
import org.springframework.util.Assert;

@Configuration
public class ApplicationContextHolder implements ApplicationContextAware {

    //静态声明context
    private static ApplicationContext applicationContext = null;

    public static ApplicationContext getApplicationContext() {
        assertContextNotNull();
        return applicationContext;
    }

    //检查ApplicationContext是否为null
    private static void assertContextNotNull() {
        Assert.notNull(applicationContext, "Context is null");
    }

    //按名称返回容器中的Bean
    public static <T> T getBean(String name) {
        assertContextNotNull();
        return (T) applicationContext.getBean(name);
    }

    //按类型返回容器中的Bean
    public static <T> T getBean(Class<T> clazz) {
        assertContextNotNull();
        return applicationContext.getBean(clazz);
    }

    //需要实现的方法，注入Context
    //Springboot会自动给实现ApplicationContextAware接口的Bean注入Context
    @Override
    public void setApplicationContext(@NotNull ApplicationContext applicationContext) throws BeansException {
        ApplicationContextHolder.applicationContext = applicationContext;
    }
}

//测试
@Resource
private DmService dmService;
log.info("{}", dmService == ApplicationContextHolder.getBean("dmService"));
