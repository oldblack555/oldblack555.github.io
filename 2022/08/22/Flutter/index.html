Flutter动画

数值变化

//动画控制器
late AnimationController _animationController;
//数值动画
late Animation<double> _valueAnimation;

@override
void initState() {
    //初始化动画控制器
    _animationController = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 1),
    );
    //初始化数值动画，确定数值变化范围，以及添加事件监听刷新UI
    _valueAnimation = Tween<double>(begin: 0.0, end: 0.75)
        .animate(_animationController)
      ..addListener(() {
        setState(() {});
      });
    //启动动画
    _animationController.forward();
    super.initState();
}

组件旋转动画

//动画控制器
late AnimationController _animationController;

@override
void initState() {
    super.initState();
    //初始哈控制器，设置动画时长
    _animationController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 300),
      upperBound: 0.5,
    );
}

//旋转组件，包裹需要旋转的组件，设置旋转角度
RotationTransition(
    turns: Tween(
        begin: 0.0,
        end: 0.5,
    ).animate(_animationController),
        child: Icon(
        defaultRightArrow,
        size: 20,
        color: widget.selectColor,
    ),
),

//点击事件逻辑，启动动画执行
if (widget.isShowNextMenu) {
    _animationController.reverse(from: 0.5);
} else {
    _animationController.forward(from: 0.0);
}
