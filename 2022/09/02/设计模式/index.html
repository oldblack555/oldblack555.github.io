工厂模式

如何实现一个两个数简单的计算功能？对于刚接触编程的人会选择下面的思路：

  1. 输入数字一
  2. 输入操作符
  3. 输入操作数二
  4. 判断操作符号
  5. 根据符号计算两数结果

但是，对于以上的思路很难扩展，如新增开方操作，这样需要在原有代码上修改大量代码和逻辑，同时也不便于维护。为此，引入面向对象的思想，将计算功能抽象，思想如下：

  1. 定义抽象计算类，封装操作数，定义计算结果的抽象方法。
  2. 定义不同操作类，继承抽象计算类，使得操作数和计算方法复用。
/**
 * 简单计算功能封装，定义操作数和计算方法，
 * 由子类继承该抽象类，实现不同的计算功能
 * @author oldblack
 */
public abstract class BaseOperation {

    private double num1 = 0;
    private double num2 = 0;



    public double getNum1() {
        return num1;
    }

    public void setNum1(double num1) {
        this.num1 = num1;
    }

    public double getNum2() {
        return num2;
    }

    public void setNum2(double num2) {
        this.num2 = num2;
    }

    /**
     * 运算并返回相应运算操作的结果
     * @return 运算结果
     */
    public abstract double result();
}

// 加法的具体实现类
import com.dp.factory.BaseOperation;

/**
 * 重写父类计算方法
 * 定义加法计算
 * @author oldblack
 */
public class OperationAdd extends BaseOperation {
    @Override
    public double result() {
        return getNum1() + getNum2();
    }
}


// 除法的具体实现类
import com.dp.factory.BaseOperation;
import com.dp.factory.config.ErrorMsg;

/**
 * 重写父类计算方法
 * 定义除法运算
 * @author oldblack
 */
public class OperationDivision extends BaseOperation {
    @Override
    public double result() {
        //检查是否除零运算
        assert getNum2() != 0 : ErrorMsg.DIVISION_ERROR.getMsg();
        return getNum1() / getNum2();
    }
}

面向对象的三个基本特性：封装、继承、多态，通过上述思想实现的简单计算，将运算的操作公共特性封装为基类，不同子类通过继承重写抽象方法，实现基类的多态，这样的好处在于，新增减法运算时，只需定义子类重写result方法实现相应的功能，提高代码复用。那么，又出现了新的问题：如何创建这些操作对象，不至于每个对象都去创建一次来实现计算功能。为此，引入工厂模式，工厂根据所需返回不同的计算对象，代码简单如下：

/**
 * 枚举计算功能，加减乘除等
 * @author oldblack
 */
public enum OperationEnum {
    //加法
    ADD,
    //减法
    SUBTRACT,
    //乘法
    MULTIPLICATION,
    //除法
    DIVISION,
    ;
}


import com.dp.factory.config.OperationEnum;
import com.dp.factory.operation.OperationAdd;
import com.dp.factory.operation.OperationDivision;
import com.dp.factory.operation.OperationMulti;
import com.dp.factory.operation.OperationSub;

/**
 * 简单的工厂实现
 * 返回不同的计算功能，通过枚举返回相应对象
 *
 * @author oldblack
 */
public class OperationFactory {
    
    public static BaseOperation build(OperationEnum operation) {
        //使用父类声明对象，用子类实例化对象，使得result方法为子类重写的方法
        BaseOperation baseOperation;
        switch (operation) {
            case SUBTRACT:
                baseOperation = new OperationSub();
                break;
            case MULTIPLICATION:
                baseOperation = new OperationMulti();
                break;
            case DIVISION:
                baseOperation = new OperationDivision();
                break;
            default:
                baseOperation = new OperationAdd();
        }
        return baseOperation;
    }

}

测试代码如下：

import com.dp.factory.BaseOperation;
import com.dp.factory.OperationFactory;
import com.dp.factory.config.OperationEnum;
import org.junit.jupiter.api.Test;

public class FactoryTest {

    @Test
    public void operationAddTest() {
        BaseOperation operationAdd = OperationFactory.build(OperationEnum.ADD);
        operationAdd.setNum1(20);
        operationAdd.setNum2(10);
        System.out.println(operationAdd.result());
    }
    //结果：30.0
    

    @Test
    public void operationSubtractTest(){
        BaseOperation operationSub = OperationFactory.build(OperationEnum.SUBTRACT);
        operationSub.setNum1(20);
        operationSub.setNum2(10);
        System.out.println(operationSub.result());
    }
    //结果：10.0

    @Test
    public void operationMultiplicationTest(){
        BaseOperation operationMulti = OperationFactory.build(OperationEnum.MULTIPLICATION);
        operationMulti.setNum1(20);
        operationMulti.setNum2(10);
        System.out.println(operationMulti.result());
    }
    //结果：200.0
    
    @Test
    public void operationDivisionTest(){
        BaseOperation operationDivision = OperationFactory.build(OperationEnum.DIVISION);
        operationDivision.setNum1(20);
        operationDivision.setNum2(10);
        System.out.println(operationDivision.result());
    }
    
    //结果：2.0

    @Test
    public void operationDivisionZeroTest(){
        BaseOperation operationDivision = OperationFactory.build(OperationEnum.DIVISION);
        operationDivision.setNum1(20);
        operationDivision.setNum2(0);
        System.out.println(operationDivision.result());
    }
    //结果：java.lang.AssertionError: 除零错误

}

通过工厂模式，我们只需要调用result方法获取计算结果即可，工厂会根据需求返回对应的计算实例，无需考虑具体具体实现，因为子类已经复写计算过程，父类负责返回结果即可，工厂负责实例化对应的子类。工厂模式极大减少业务调用时的逻辑判断，调用这只需按需使用即可。工厂模式核心在于生产和返回实例，不进行其他的业务操作。不足之处在于工厂类中存在条件判断，没有解决判断的问题（解决办法：工厂方法模式）。
