介绍

Slf4j是一个日志接口，不提供具体的实现类。实用Slf4j作为日志框架则需要和一个具体的实现类结合实用，否则无效。下图为Slf4j和对应的实现类关系。

Slf4j使用

使用slf4j-api和slf4j-simple组合

//导入依赖
implementation("org.slf4j:slf4j-simple:2.0.0")
implementation("org.slf4j:slf4j-api:2.0.0")
    
//slf4j使用
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import redis.clients.jedis.Jedis;

public class Slf4jDome {

    public static void main(String[] args) {
        Logger logger = LoggerFactory.getLogger(Slf4jDome.class);
        logger.info("Hello World");
    }
}

使用log4j

  • 导入依赖
implementation("log4j:log4j:1.2.17")
  • 在项目资源文件（根目录）配置log4j.properties
#日志配置
log4j.rootLogger = DEBUG,stdout
#,file

#控制台输出
log4j.appender.stdout=org.apache.log4j.ConsoleAppender
log4j.appender.stdout.Target=System.out
log4j.appender.stdout.Threshold=DEBUG
log4j.appender.stdout.layout=org.apache.log4j.PatternLayout
log4j.appender.stdout.layout.ConversionPattern=%-d{yyyy-MM-dd HH:mm:ss}[ %p ]%m%n

#所有文件输出
#log4j.appender.file = org.apache.log4j.FileAppender
#log4j.appender.file.File = D:/logs/log.log
#log4j.appender.file.Encoding=UTF-8
#log4j.appender.file.name = fileLogDemo
#log4j.appender.file.Threshold=DEBUG
#log4j.appender.file.layout=org.apache.log4j.PatternLayout
#log4j.appender.file.layout.ConversionPattern=%-d{yyyy-MM-dd HH:mm:ss} [%c {Num}] [%l] [ %t:%r ] - [ %p ]  %m%n
#log4j.appender.file.append = true
  • 创建Logger
import org.apache.log4j.Logger;

@Test
public void log4jTest(){
    Logger logger = Logger.getLogger(Slf4jDome.class);
    logger.info("Hello World");
}
